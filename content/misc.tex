\subsection{Inverse of a Perturbed Matrix}

\begin{itemize} 
\item $(I + UV)^{-1} = I - U(I + VU)^{-1} V$.
\item $(A + UCV)^{-1} = A^{-1} - A^{-1} U (C^{-1} + VA^{-1} U)^{-1} VA^{-1}$
\item $(A + uv^T)^{-1} = A^{-1} - (A^{-1} uv^T A^{-1})/(1 + v^T A^{-1} u)$
\item $v^T A^{-1} u = v^T x$, где $x$ --- решение $Ax = u$.
\end{itemize}
